## 🔑 步骤1：获取API密钥

<Steps>
  <Step title="注册账户">
    访问 [EvoLink](https://evolink.ai/) 并注册新账户
  </Step>
  <Step title="创建密钥">
    在控制台中点击"API密钥"，创建一个新的密钥
  </Step>
  <Step title="保存密钥">
    复制生成的API密钥，格式为：`sk-evo-xxxxxxxxxxxxxxxx`
  </Step>
</Steps>

<Warning>
  请妥善保管您的API密钥，不要在客户端代码中硬编码或公开分享。

## 💻 步骤3：SDK集成

EvoLink API兼容所有OpenAI SDK，您可以直接使用现有的SDK库。

<Tabs>
  <Tab title="Python">
    ```bash
    # 安装OpenAI SDK
    pip install openai
    ```

    ```python
    import openai
    
    # 配置EvoLink API
    client = openai.OpenAI(
        api_key="sk-evo-your-api-key",
        base_url="https://api.evolink.ai"
    )
    
    # 文本对话
    def chat_example():
        response = client.chat.completions.create(
            model="gpt-4o",
            messages=[
                {"role": "system", "content": "你是一个有用的AI助手"},
                {"role": "user", "content": "请介绍一下Python的优势"}
            ],
            max_tokens=500
        )
        print(response.choices[0].message.content)
    
    # 图像生成
    def image_example():
        response = client.images.generate(
            model="dall-e-3",
            prompt="A serene lake with mountains in the background, digital art",
            size="1024x1024",
            quality="hd",
            n=1
        )
        print(f"Generated image: {response.data[0].url}")
    
    # 语音转文字
    def transcription_example():
        with open("speech.mp3", "rb") as audio_file:
            response = client.audio.transcriptions.create(
                model="whisper-1",
                file=audio_file,
                language="zh"
            )
        print(f"Transcription: {response.text}")
    
    # 运行示例
    if __name__ == "__main__":
        chat_example()
        image_example()
        transcription_example()
    ```

        quality: 'hd'
      });
      
      console.log(`Generated image: ${image.data[0].url}`);

      console.log(`Transcription: ${transcription.text}`);
    }
    
    // 运行示例
    async function main() {
      await chatExample();
      await imageExample();
      await transcriptionExample();
    }
    
    main().catch(console.error);

    import com.theokanning.openai.OpenAiService;
    import com.theokanning.openai.completion.chat.*;
    import com.theokanning.openai.image.*;
    import java.util.Arrays;
    
    public class EvoLinkExample {
        public static void main(String[] args) {
            // 配置EvoLink API
            OpenAiService service = new OpenAiService("sk-evo-your-api-key");
            service.setBaseUrl("https://api.evolink.ai");
            
            // 文本对话
            chatExample(service);
            
            // 图像生成
            imageExample(service);
        }
        
        public static void chatExample(OpenAiService service) {
            ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-4o")
                .messages(Arrays.asList(
                    new ChatMessage(ChatMessageRole.SYSTEM.value(), "你是一个Java技术专家"),
                    new ChatMessage(ChatMessageRole.USER.value(), "Java的主要优势是什么？")
                ))
                .maxTokens(500)
                .build();
                
            ChatCompletionResult result = service.createChatCompletion(request);
            System.out.println(result.getChoices().get(0).getMessage().getContent());
        }
        
        public static void imageExample(OpenAiService service) {
            CreateImageRequest request = CreateImageRequest.builder()
                .model("dall-e-3")
                .prompt("A beautiful Java coffee cup with steam, photorealistic")
                .size("1024x1024")
                .quality("hd")
                .n(1)
                .build();
                
            ImageResult result = service.createImage(request);
            System.out.println("Generated image: " + result.getData().get(0).getUrl());
        }
    }

    import (
        "context"
        "fmt"

        // 配置EvoLink API

        client := openai.NewClientWithConfig(config)
        
        // 文本对话示例

        // 图像生成示例

    func chatExample(client *openai.Client) {
        resp, err := client.CreateChatCompletion(
            context.Background(),
            openai.ChatCompletionRequest{
                Model: openai.GPT4,
                Messages: []openai.ChatCompletionMessage{
                    {
                        Role:    openai.ChatMessageRoleSystem,
                        Content: "你是一个Go语言专家",
                    },
                    {
                        Role:    openai.ChatMessageRoleUser,
                        Content: "Go语言有什么独特的优势？",
                    },
                },
                MaxTokens: 500,
            },
        )
        
        if err != nil {
            fmt.Printf("ChatCompletion error: %v\n", err)
            return
        }
        
        fmt.Println(resp.Choices[0].Message.Content)
    }
    
    func imageExample(client *openai.Client) {
        resp, err := client.CreateImage(
            context.Background(),
            openai.ImageRequest{
                Model:   openai.CreateImageModelDallE3,
                Prompt:  "A gopher programming Go code, cartoon style",
                Size:    openai.CreateImageSize1024x1024,
                Quality: openai.CreateImageQualityHD,
                N:       1,
            },
        )
        
        if err != nil {
            fmt.Printf("Image creation error: %v\n", err)
            return
        }
        
        fmt.Println("Generated image:", resp.Data[0].URL)
    }

## 🔧 步骤4：高级配置

### 流式响应

对于聊天应用，您可以启用流式响应以获得更好的用户体验：

<Tabs>
  <Tab title="Python">
    ```python
    # 流式聊天响应
    stream = client.chat.completions.create(
        model="gpt-4o",
        messages=[
            {"role": "user", "content": "请详细介绍人工智能的发展历程"}
        ],
        stream=True
    )
    
    for chunk in stream:
        if chunk.choices[0].delta.content is not None:
            print(chunk.choices[0].delta.content, end="")
    ```
  </Tab>
  <Tab title="Node.js">
    ```javascript
    const stream = await client.chat.completions.create({
      model: 'gpt-4o',
      messages: [{ role: 'user', content: '请详细介绍机器学习的基本概念' }],
      stream: true,
    });
    
    for await (const chunk of stream) {
      if (chunk.choices[0]?.delta?.content) {
        process.stdout.write(chunk.choices[0].delta.content);
      }
    }
    ```
  </Tab>
</Tabs>

### 错误处理

始终添加适当的错误处理：

```python
try:
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=[{"role": "user", "content": "Hello"}]
    )
    print(response.choices[0].message.content)
    
except openai.RateLimitError:
    print("API请求频率过高，请稍后重试")
except openai.APIError as e:
    print(f"API错误: {e}")
except Exception as e:
    print(f"其他错误: {e}")
```
